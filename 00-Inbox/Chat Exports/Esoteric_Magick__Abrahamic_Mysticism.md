##### **B. Define API Endpoints (`backend/app.py`)**
```python
from fastapi import FastAPI
from pydantic import BaseModel
import supabase

app = FastAPI()

# Initialize Supabase client (use env vars later)
supabase_client = supabase.create_client(
    "YOUR_SUPABASE_URL",
    "YOUR_SUPABASE_KEY"
)

class TextUpload(BaseModel):
    title: str
    content: str
    tradition: str  # e.g., "Islamic", "Kabbalah"

@app.post("/upload-text")
async def upload_text(text: TextUpload):
    """Upload esoteric texts to Supabase."""
    data = supabase_client.table("esoteric_texts").insert({
        "title": text.title,
        "content": text.content,
        "tradition": text.tradition
    }).execute()
    return {"status": "success", "id": data.data[0]["id"]}

@app.get("/texts/{tradition}")
async def get_texts(tradition: str):
    """Fetch texts by tradition."""
    data = supabase_client.table("esoteric_texts").select("*").eq("tradition", tradition).execute()
    return data.data
```
